> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the lexical-gated dense scoring in `rerank=rrf` with true parallel hybrid retrieval (lexical-top-K ∪ dense-top-K corpus-wide, filter-aware), and deliver the benchmark harness + dataset that C-β will use to pick a cross-encoder.

**Tech Stack:** Python 3.12, FastAPI, asyncpg, Pydantic 2, pgvector 0.7+ with HNSW, pytest, ruff, mypy strict, Codex CLI (for benchmark generation).

**Source spec:** `docs/superpowers/specs/2026-05-12-ranking-architecture-redesign-design.md`

**Out of scope for this plan:** C-β (model bake-off) and C-γ (cross-encoder mode ship) get their own plans after C-α merges. This plan delivers the recall-ceiling fix and the benchmark infrastructure — nothing more.

## Task 2: Stratified chapter worklist for benchmark generation

**Files:**

- [ ] **Step 1: Write the worklist-builder script**

Create `scripts/build_bench_worklist.py`:

```python
"""Build a stratified chapter worklist for benchmark seed generation.

Output: tests/fixtures/ranking_bench_worklist.json with ~30 chapters
stratified across:
  - section presence (must have management + diagnosis + molecular_genetics
    populated to ensure variety in seeded passages)
  - chapter age (recent + old quarter, plus 2 'middle' quarters)
  - gene popularity (well-known vs rare, approximated by gene_symbols length
    in chapter row + presence in OMIM-flagged lists)

Selection: deterministic given a seed (default 42) so the worklist is
reproducible.
"""
from __future__ import annotations
import argparse
import json
import random
import sys
from pathlib import Path

from genereview_link.config import get_settings
from genereview_link.db.pool import build_pool
import asyncio


async def fetch_candidate_chapters(min_sections: int = 3):
    pool = await build_pool(get_settings().database)
    async with pool.acquire() as conn:
        rows = await conn.fetch("""
            select
                c.nbk_id,
                c.title,
                c.last_updated,
                c.gene_symbols,
                count(distinct p.chapter_section) filter (where p.passage_count > 0)
                    as populated_sections
            from genereview_chapters c
            join (
                select nbk_id, chapter_section, count(*) as passage_count
                from genereview_passages
                group by nbk_id, chapter_section
            ) p on p.nbk_id = c.nbk_id
            group by c.nbk_id, c.title, c.last_updated, c.gene_symbols
            having count(distinct p.chapter_section) filter (where p.passage_count > 0) >= $1
        """, min_sections)
    await pool.close()
    return rows


def stratify(rows, n_per_bucket: int = 8, seed: int = 42):
    rng = random.Random(seed)
    # Bucket by chapter_last_updated quartile (None goes to "unknown" bucket).
    sorted_with_dates = sorted(
        [r for r in rows if r["last_updated"]],
        key=lambda r: r["last_updated"]
    )
    q = len(sorted_with_dates) // 4
    buckets = {
        "oldest_q": sorted_with_dates[:q],
        "old_mid": sorted_with_dates[q:2*q],
        "new_mid": sorted_with_dates[2*q:3*q],
        "newest_q": sorted_with_dates[3*q:],
    }
    selected = []
    for name, bucket in buckets.items():
        if not bucket:
            continue
        sample = rng.sample(bucket, min(n_per_bucket, len(bucket)))
        for r in sample:
            selected.append({
                "nbk_id": r["nbk_id"],
                "title": r["title"],
                "last_updated": str(r["last_updated"]) if r["last_updated"] else None,
                "gene_symbols": r["gene_symbols"],
                "bucket": name,
            })
    return selected


async def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--seed", type=int, default=42)
    ap.add_argument("--n-per-bucket", type=int, default=8)
    ap.add_argument("--output", type=Path, default=Path("tests/fixtures/ranking_bench_worklist.json"))
    args = ap.parse_args()

    rows = await fetch_candidate_chapters()
    worklist = stratify(rows, n_per_bucket=args.n_per_bucket, seed=args.seed)
    args.output.write_text(json.dumps(worklist, indent=2))
    print(f"wrote {len(worklist)} chapters to {args.output}")


if __name__ == "__main__":
    asyncio.run(main())
```

- [ ] **Step 1: Write the orchestrator script**

Create `scripts/generate_ranking_bench.py`:

```python
"""Orchestrator: iterate worklist, fetch seed passages, dispatch Codex,
write silver entries to tests/fixtures/ranking_bench.jsonl.

Codex is invoked via subprocess against the locally-installed codex-cli.
For each chapter, pick up to N=5 substantive passages (passage_role='evidence',
char_count between 300 and 2000) across the populated sections, call Codex
once per passage with the template at tests/fixtures/ranking_bench_generation_prompt.md.

Resumes from a partial output file (skips already-processed passage_ids).
"""
from __future__ import annotations
import argparse
import asyncio
import json
import subprocess
from pathlib import Path
from typing import Any

from genereview_link.config import get_settings
from genereview_link.db.pool import build_pool


PROMPT_PATH = Path("tests/fixtures/ranking_bench_generation_prompt.md")
SEEDS_PER_CHAPTER = 5


async def fetch_seed_passages(nbk_id: str):
    pool = await build_pool(get_settings().database)
    async with pool.acquire() as conn:
        rows = await conn.fetch("""
            select passage_id, chapter_section, heading_path, text, char_count
            from genereview_passages
            where nbk_id = $1
              and passage_role = 'evidence'
              and char_count between 300 and 2000
            order by char_count desc
            limit $2
        """, nbk_id, SEEDS_PER_CHAPTER)
    await pool.close()
    return rows


def render_prompt(template: str, chapter_title: str, row: dict) -> str:
    return (template
        .replace("{NBK_ID}", row["passage_id"].split(":")[0])
        .replace("{CHAPTER_TITLE}", chapter_title)
        .replace("{SECTION}", row["chapter_section"])
        .replace("{HEADING_PATH}", row["heading_path"] or "")
        .replace("{PASSAGE_ID}", row["passage_id"])
        .replace("{PASSAGE_TEXT}", row["text"])
    )


def run_codex(prompt: str) -> dict:
    """Run codex task non-interactively, parse JSON output."""
    result = subprocess.run(
        ["codex", "task", "-c", "model_reasoning_effort=medium", "--non-interactive"],
        input=prompt, capture_output=True, text=True, check=True, timeout=120,
    )
    return json.loads(result.stdout)


def already_processed(jsonl_path: Path) -> set[str]:
    if not jsonl_path.exists():
        return set()
    seen = set()
    for line in jsonl_path.read_text().splitlines():
        if not line.strip() or line.startswith("#"):
            continue
        try:
            seen.add(json.loads(line)["expected_top1_passage_id"])
        except (json.JSONDecodeError, KeyError):
            continue
    return seen


async def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--worklist", type=Path,
                    default=Path("tests/fixtures/ranking_bench_worklist.json"))
    ap.add_argument("--output", type=Path,
                    default=Path("tests/fixtures/ranking_bench.jsonl"))
    args = ap.parse_args()

    worklist = json.loads(args.worklist.read_text())
    template = PROMPT_PATH.read_text()
    seen = already_processed(args.output)
    print(f"resuming: {len(seen)} passages already processed")

    with args.output.open("a") as out:
        for chapter in worklist:
            seeds = await fetch_seed_passages(chapter["nbk_id"])
            for seed in seeds:
                if seed["passage_id"] in seen:
                    continue
                prompt = render_prompt(template, chapter["title"], dict(seed))
                try:
                    response = run_codex(prompt)
                except Exception as e:
                    print(f"  codex error on {seed['passage_id']}: {e}", flush=True)
                    continue

                if response.get("skip"):
                    print(f"  skip {seed['passage_id']}: {response.get('reason')}", flush=True)
                    continue

                for q in response.get("queries", []):
                    entry = {
                        "query": q["query"],
                        "expected_top1_passage_id": seed["passage_id"],
                        "expected_top5_passage_ids": [seed["passage_id"]],
                        "status": "silver",
                        "source": "codex_generated_2026-05-12",
                        "intent": q.get("intent", "other"),
                        "section": seed["chapter_section"],
                        "chapter_nbk_id": chapter["nbk_id"],
                        "notes": f"Codex-generated from {seed['passage_id']}, style={q.get('style')}.",
                    }
                    out.write(json.dumps(entry) + "\n")
                    out.flush()
                print(f"  ok {seed['passage_id']}: {len(response.get('queries', []))} queries", flush=True)


if __name__ == "__main__":
    asyncio.run(main())
```

## Task 5: Auto-validate silver entries against non-CE retrieval modes

**Files:**
- Create: `scripts/validate_ranking_bench.py`
- Modify: `tests/fixtures/ranking_bench.jsonl` (annotate validation bucket)

- [ ] **Step 1: Write the validator**

Create `scripts/validate_ranking_bench.py`:

```python
"""Run each silver entry through the live MCP across non-CE retrieval modes
(lexical, rrf-CURRENT-PRE-UPGRADE, off), assign bucket A/B/C, write back to
ranking_bench.jsonl with a 'validation_bucket' field.

Bucket A: all 3 modes return gold in top-5 → silver confirmed.
Bucket B: modes disagree on top-5 → flag for SME spot-check.
Bucket C: no mode returns gold in top-50 → either hard-recall test or bogus.

Note: at this point the parallel-retrieval rrf is NOT yet implemented.
This task runs against the CURRENT (gated) rrf, which gives us the baseline
to measure against after C-α's main code change lands.
"""
import argparse
import asyncio
import json
import os
from pathlib import Path

import httpx


async def query_mode(client, q, mode, gold_id):
    """Returns (in_top5: bool, in_top50: bool, top1_id: str | None)."""
    r = await client.get("/passages/search", params={
        "q": q, "rerank": mode, "mode": "ids_only", "limit": 50,
    })
    r.raise_for_status()
    rows = r.json().get("results", [])
    ids = [row["passage_id"] for row in rows]
    return (
        gold_id in ids[:5],
        gold_id in ids[:50],
        ids[0] if ids else None,
    )


async def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--input", type=Path,
                    default=Path("tests/fixtures/ranking_bench.jsonl"))
    ap.add_argument("--base-url", default=os.environ.get("MCP_BASE_URL", "http://127.0.0.1:8765"))
    args = ap.parse_args()

    entries = [json.loads(l) for l in args.input.read_text().splitlines()
               if l.strip() and not l.startswith("#")]

    async with httpx.AsyncClient(base_url=args.base_url, timeout=30.0) as client:
        for e in entries:
            gold = e["expected_top1_passage_id"]
            modes = ["lexical", "rrf", "off"]
            results = {}
            for m in modes:
                try:
                    results[m] = await query_mode(client, e["query"], m, gold)
                except Exception as ex:
                    results[m] = (False, False, f"error: {ex}")

            top5 = [r[0] for r in results.values() if isinstance(r[0], bool)]
            top50 = [r[1] for r in results.values() if isinstance(r[1], bool)]

            if all(top5):
                e["validation_bucket"] = "A"
            elif any(top50):
                e["validation_bucket"] = "B"
            else:
                e["validation_bucket"] = "C"
            e["validation_detail"] = {m: {"top1": r[2]} for m, r in results.items()}

    # Rewrite the file with bucket annotations.
    with args.input.open("w") as f:
        for e in entries:
            f.write(json.dumps(e) + "\n")

    from collections import Counter
    print("buckets:", Counter(e["validation_bucket"] for e in entries))


if __name__ == "__main__":
    asyncio.run(main())
```

Create `scripts/bench_ranking.py`:

```python
"""Run each ranking_bench.jsonl entry against the live MCP across each retrieval mode,
compute P@1, MRR@5, Recall@5, and print a comparison table.

Hard gates enforced:
  - must_not_regress + regression_kind=exact-symbol-anchor: any top-1 change fails the run.
  - must_change: at least 1 entry's top-1 must improve, or the run fails.

Honors --regression-kind-strict flag (default true) to disable the
must_not_regress gate for the pending-improvement subset.

Output: stdout table + optional JSON dump via --json-out.
"""
import argparse
import asyncio
import json
import os
import sys
from collections import defaultdict
from pathlib import Path

import httpx


MODES = ["lexical", "rrf", "off"]  # rerank=ce added later when C-γ ships


def reciprocal_rank(ids: list[str], gold: str) -> float:
    for i, p in enumerate(ids):
        if p == gold:
            return 1.0 / (i + 1)
    return 0.0


async def run_one(client, entry, mode):
    r = await client.get("/passages/search", params={
        "q": entry["query"], "rerank": mode, "mode": "ids_only", "limit": 5,
    })
    r.raise_for_status()
    ids = [row["passage_id"] for row in r.json().get("results", [])]
    gold1 = entry["expected_top1_passage_id"]
    gold5 = set(entry.get("expected_top5_passage_ids", [gold1]))
    return {
        "p_at_1": 1.0 if ids and ids[0] == gold1 else 0.0,
        "mrr_at_5": reciprocal_rank(ids, gold1),
        "recall_at_5": 1.0 if any(p in gold5 for p in ids) else 0.0,
        "top1": ids[0] if ids else None,
    }


async def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--bench", type=Path, default=Path("tests/fixtures/ranking_bench.jsonl"))
    ap.add_argument("--base-url", default=os.environ.get("MCP_BASE_URL", "http://127.0.0.1:8765"))
    ap.add_argument("--json-out", type=Path)
    args = ap.parse_args()

    entries = [json.loads(l) for l in args.bench.read_text().splitlines()
               if l.strip() and not l.startswith("#")]

    results = {m: defaultdict(list) for m in MODES}
    regressions = {m: [] for m in MODES}
    improvements = {m: [] for m in MODES}

    async with httpx.AsyncClient(base_url=args.base_url, timeout=60.0) as client:
        for entry in entries:
            for mode in MODES:
                r = await run_one(client, entry, mode)
                results[mode]["p_at_1"].append(r["p_at_1"])
                results[mode]["mrr_at_5"].append(r["mrr_at_5"])
                results[mode]["recall_at_5"].append(r["recall_at_5"])

                # Hard gates
                if entry.get("status") == "must_not_regress" and \
                        entry.get("regression_kind") == "exact-symbol-anchor":
                    if r["top1"] != entry["expected_top1_passage_id"]:
                        regressions[mode].append((entry["query"], entry["expected_top1_passage_id"], r["top1"]))

                if entry.get("status") == "must_change":
                    if r["p_at_1"] == 1.0:
                        improvements[mode].append(entry["query"])

    # Print comparison table.
    print(f"\n{'Mode':<10} {'P@1':>8} {'MRR@5':>8} {'Recall@5':>10} {'Composite':>10} {'Regressions':>12} {'Improvements':>13}")
    print("-" * 80)
    summary = {}
    for mode in MODES:
        n = len(entries)
        p1 = sum(results[mode]["p_at_1"]) / n
        mrr = sum(results[mode]["mrr_at_5"]) / n
        r5 = sum(results[mode]["recall_at_5"]) / n
        comp = (p1 + mrr + r5) / 3.0
        reg = len(regressions[mode])
        imp = len(improvements[mode])
        print(f"{mode:<10} {p1:>8.3f} {mrr:>8.3f} {r5:>10.3f} {comp:>10.3f} {reg:>12} {imp:>13}")
        summary[mode] = {
            "p_at_1": p1, "mrr_at_5": mrr, "recall_at_5": r5,
            "composite": comp, "regressions": regressions[mode],
            "improvements": improvements[mode],
        }

    # Hard gate enforcement: exact-symbol-anchor regressions fail the run.
    failed = False
    for mode in MODES:
        if regressions[mode]:
            print(f"\nFAIL: {mode} regressed {len(regressions[mode])} exact-symbol-anchor entries:")
            for q, gold, got in regressions[mode]:
                print(f"  {q}: expected {gold}, got {got}")
            failed = True

    if args.json_out:
        args.json_out.write_text(json.dumps(summary, indent=2))

    sys.exit(1 if failed else 0)


if __name__ == "__main__":
    asyncio.run(main())
```

```python
def build_dense_candidates_sql(
    *,
    embedding_table: str,
    gene: str | None,
    nbk_id: str | None,
    sections: tuple[str, ...] | None,
    heading_path_contains: str | None,
    top_k: int,
) -> tuple[str, list[object]]:
    """Build the dense-candidate SQL with filter-aware top-K.

    Strategy:
      - Single-chapter filter (nbk_id given): bypass HNSW, exact cosine over the
        small filtered set.
      - Other filters: HNSW with iterative scan + larger ef_search.
      - No filter: vanilla HNSW.

    Returns (sql, params). The first param ($1) is always the query embedding;
    subsequent params are filter values in the order: gene, nbk_id, sections, heading_path_contains.
    """
    params: list[object] = []
    param_idx = 1

    def next_param(value):
        nonlocal param_idx
        params.append(value)
        idx = param_idx
        param_idx += 1
        return f"${idx}"

    embedding_param = next_param([])  # placeholder; caller fills in the query vector
    where_clauses = []

    if gene:
        where_clauses.append(f"c.gene_symbols @> array[{next_param(gene)}]")
    if nbk_id:
        where_clauses.append(f"p.nbk_id = {next_param(nbk_id)}")
    if sections:
        where_clauses.append(f"p.chapter_section = any({next_param(list(sections))}::text[])")
    if heading_path_contains:
        where_clauses.append(f"p.heading_path ilike {next_param('%' + heading_path_contains + '%')}")

    where_sql = " and ".join(where_clauses) if where_clauses else "true"
    top_k_param = next_param(top_k)

    # If nbk_id is the only filter, use exact KNN (bypass HNSW).
    if nbk_id and not (gene or sections or heading_path_contains):
        sql = f"""
            select p.passage_id, 1 - (e.embedding <=> {embedding_param}::vector) as dense_score
            from "{embedding_table}" e
            join genereview_passages p on p.nbk_id = e.nbk_id and p.passage_id = e.passage_id
            where {where_sql}
            order by e.embedding <=> {embedding_param}::vector
            limit {top_k_param}
        """
        return sql, params

    # Otherwise: HNSW with iterative scan.
    sql = f"""
        set local hnsw.iterative_scan = 'relaxed_order';
        set local hnsw.ef_search = 200;
        select p.passage_id, 1 - (e.embedding <=> {embedding_param}::vector) as dense_score
        from "{embedding_table}" e
        join genereview_passages p on p.nbk_id = e.nbk_id and p.passage_id = e.passage_id
        join genereview_chapters c on c.nbk_id = p.nbk_id
        where {where_sql}
        order by e.embedding <=> {embedding_param}::vector
        limit {top_k_param}
    """
    return sql, params


def build_parallel_search_sql(
    *,
    query_text: str,
    query_vector: list[float],
    gene: str | None,
    nbk_id: str | None,
    sections: tuple[str, ...] | None,
    heading_path_contains: str | None,
    top_k: int,
) -> tuple[str, list[object]]:
    """Build the parallel-retrieval SQL: lexical-top-K UNION dense-top-K.

    Each branch applies the same filters. RRF fusion happens in Python
    after fetching the union.
    """
    # For test purposes here, we return a SQL skeleton that the test can
    # introspect; the production repository will use this as a building block
    # alongside the existing lexical query.
    dense_sql, dense_params = build_dense_candidates_sql(
        embedding_table="genereview_embeddings_bge384",
        gene=gene, nbk_id=nbk_id, sections=sections,
        heading_path_contains=heading_path_contains, top_k=top_k,
    )
    # Existing lexical SQL lives in Repository._lexical_candidates_sql or similar;
    # here we just compose a UNION shape for unit-test introspection.
    lexical_sql = "select passage_id, null::float as dense_score from genereview_passages limit 1"
    return (
        f"({lexical_sql}) union ({dense_sql})",
        dense_params,
    )
```

```python
"""For each filter combination, run EXPLAIN ANALYZE on the dense branch
and record the chosen plan (HNSW vs seq scan), cost, and rows-returned.

Output: markdown doc summarising which filters trigger which plans.
Spec requires this as part of C-α deliverables.
"""
import asyncio
import json
from pathlib import Path

from genereview_link.config import get_settings
from genereview_link.db.pool import build_pool
from genereview_link.retrieval.repository import build_dense_candidates_sql


FILTER_CASES = [
    {"name": "no_filter", "gene": None, "nbk_id": None, "sections": None, "heading_path_contains": None},
    {"name": "gene_only", "gene": "HFE", "nbk_id": None, "sections": None, "heading_path_contains": None},
    {"name": "nbk_id_only", "gene": None, "nbk_id": "NBK1247", "sections": None, "heading_path_contains": None},
    {"name": "section_only", "gene": None, "nbk_id": None, "sections": ("management",), "heading_path_contains": None},
    {"name": "gene_plus_section", "gene": "BRCA1", "nbk_id": None, "sections": ("management",), "heading_path_contains": None},
    {"name": "heading_path_only", "gene": None, "nbk_id": None, "sections": None, "heading_path_contains": "Prevention"},
]


async def main():
    pool = await build_pool(get_settings().database)
    output: list[dict] = []
    dummy_vector = [0.0] * 384

    for case in FILTER_CASES:
        sql, params = build_dense_candidates_sql(
            embedding_table="genereview_embeddings_bge384",
            top_k=200, **{k: v for k, v in case.items() if k != "name"},
        )
        params[0] = dummy_vector
        explain_sql = "EXPLAIN (ANALYZE, FORMAT JSON, BUFFERS) " + sql
        async with pool.acquire() as conn:
            plan = await conn.fetchval(explain_sql, *params)
        plan_node = plan[0]["Plan"]
        output.append({
            "case": case["name"],
            "node_type": plan_node["Node Type"],
            "total_cost": plan_node["Total Cost"],
            "actual_time_ms": plan_node["Actual Total Time"],
            "rows_returned": plan_node["Plan Rows"],
            "plan_json": plan,
        })

    Path("c_alpha_hnsw_plans.json").write_text(json.dumps(output, indent=2))
    await pool.close()
    print(f"wrote {len(output)} plan entries to c_alpha_hnsw_plans.json")
    for r in output:
        print(f"  {r['case']:<25} {r['node_type']:<35} {r['actual_time_ms']:>8.2f}ms  ({r['rows_returned']} rows)")


if __name__ == "__main__":
    asyncio.run(main())
```
